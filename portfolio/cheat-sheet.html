Accessing Nested Hash Elements

nested_hash = {
               bet: {
                     "you" => ["c","a","n"],
                     not:  {
                            reach: "THIS"
                           }
                    }
              }


puts nested_hash[:bet][:not][:reach]
# Output
# THIS


nested_hash.each do |first_layer, first_value|
  first_value.each do |second_layer, second_value|
    second_value.each do |third_layer, third_value|
      puts third_value
    end
  end
end
# Output
# THIS
